# Define sigmod function
def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# Define the derivate of the sigmoid function
def d_sigmoid(z):
    return sigmoid(z) * (1 - sigmoid(z))

# Define backpropagation function with a hidden layer
def backprop(X, y, h=5, epochs=500, eta=0.1):
    """
    Trains a 1-hidden-layer MLP (with h neurons) using backpropagation.
    
    Parameters:
    -----------
    X : array of shape (n_samples, n_features)
        Input data.
    y : array of shape (n_samples,)
        Binary labels in {0, 1}.
    h : int
        Number of neurons in the hidden layer.
    epochs : int
        Number of training epochs.
    eta : float
        Learning rate.
    
    Returns:
    --------
    W1, b1, W2, b2 : ndarrays
        Trained weights and biases for the hidden and output layers.
    errors : list of floats
        Evolution of the error (loss) over epochs.
    """
    np.random.seed(43)

    n_samples, n_features = X.shape # 100, 2
    y = y.reshape(-1, 1) # 100, 1 (column vector)

    # Initialize weights
    # Hidden layer: W1 is (n_features x h), b1 is (1 x h)
    W1 = np.random.randn(n_features, h) # (2 features, 5 neurons) where h is the hidden layer
    b1 = np.random.randn(1, h) # 1 row, 5 columns; bias for each neuron

    # Output layer: W2 is (h x 1), b2 is (1 x 1)
    W2 = np.random.randn(h, 1) # 5 neurons, 1 output; from hidden layer to output layer
    b2 = np.random.randn(1, 1) # bias for output layer

    errors = []

    for epoch in range(epochs):
        # ---- Forward Pass ----
        z1 = np.dot(X, W1) + b1 
        a1 = sigmoid(z1) # activation of hidden layer
        
        z2 = np.dot(a1, W2) + b2 
        a2 = sigmoid(z2) # activation of output layer

        # Cost function: MSE
        error = np.mean((a2 - y) ** 2)
        errors.append(error)
        
        # ---- Backward Pass (Backpropagation) ----
        dE_dz2 = (a2 - y) * d_sigmoid(z2) # error gradient
        
        dE_da1 = dE_dz2.dot(W2.T) # hidden layer contribution to the error
        dE_dz1 = dE_da1 * d_sigmoid(z1)

        # Update weights
        dW2 = np.dot(a1.T, dE_dz2)
        db2 = np.sum(dE_dz2, axis=0, keepdims=True)

        dW1 = np.dot(X.T, dE_dz1)
        db1 = np.sum(dE_dz1, axis=0, keepdims=True)

        W2 -= eta * dW2
        b2 -= eta * db2
        W1 -= eta * dW1
        b1 -= eta * db1
        
    return W1, b1, W2, b2, errors

Hyperparameter Tuning via Grid Search

In this code block, I perform a grid search over two hyperparameters:

  • The number of neurons in the hidden layer (h).
  • The learning rate (eta).

For each combination of h and eta, I train the network for a fixed number of epochs (1000) and record the final training error. I keep track of the best (lowest) error seen so far and update the “best configuration” accordingly. By iterating through all possible parameter pairs, I identify the optimal setup (in terms of minimal final error) for the model.

This function trains a single-hidden-layer MLP using backpropagation with momentum. For each epoch, it performs a forward pass to compute the output and error, then a backward pass to compute gradients, and finally updates the weights and biases by adding a momentum term. The function returns the trained parameters and the error history over all epochs. The momentum term helps speed up convergence and avoid getting stuck in local minima.

def backprop(X, y, h=5, epochs=500, eta=0.1, momentum=0.9):
    """
    Trains a single hidden-layer MLP (with h neurons) using backpropagation with momentum.
    
    Parameters:
    -----------
    X : array of shape (n_samples, n_features)
        Input data.
    y : array of shape (n_samples,)
        Binary labels in {0, 1}.
    h : int
        Number of neurons in the hidden layer.
    epochs : int
        Number of training epochs.
    eta : float
        Learning rate.
    momentum : float
        Momentum coefficient.
    
    Returns:
    --------
    W1, b1, W2, b2 : arrays
        Trained weights and biases for the hidden and output layers.
    errors : list of floats
        Evolution of the error (loss) over epochs.
    """
    np.random.seed(43)

    n_samples, n_features = X.shape
    y = y.reshape(-1, 1)

    # Initialize weights and biases
    W1 = np.random.randn(n_features, h)
    b1 = np.random.randn(1, h)
    W2 = np.random.randn(h, 1)
    b2 = np.random.randn(1, 1)

    # Initialize velocity terms (for momentum) to zeros
    V_W1 = np.zeros_like(W1)
    V_b1 = np.zeros_like(b1)
    V_W2 = np.zeros_like(W2)
    V_b2 = np.zeros_like(b2)

    errors = []

    for epoch in range(epochs):
        # ---- Forward Pass ----
        z1 = np.dot(X, W1) + b1 
        a1 = sigmoid(z1)
        z2 = np.dot(a1, W2) + b2 
        a2 = sigmoid(z2)

        # Compute the error (Mean Squared Error)
        error = np.mean((a2 - y) ** 2)
        errors.append(error)
        
        # ---- Backward Pass (Backpropagation) ----
        dE_dz2 = (a2 - y) * d_sigmoid(z2)
        dE_da1 = np.dot(dE_dz2, W2.T)
        dE_dz1 = dE_da1 * d_sigmoid(z1)

        dW2 = np.dot(a1.T, dE_dz2)
        db2 = np.sum(dE_dz2, axis=0, keepdims=True)

        dW1 = np.dot(X.T, dE_dz1)
        db1 = np.sum(dE_dz1, axis=0, keepdims=True)

        # ---- Update with Momentum ----
        V_W2 = momentum * V_W2 - eta * dW2
        V_b2 = momentum * V_b2 - eta * db2
        V_W1 = momentum * V_W1 - eta * dW1
        V_b1 = momentum * V_b1 - eta * db1

        W2 += V_W2
        b2 += V_b2
        W1 += V_W1
        b1 += V_b1

    return W1, b1, W2, b2, errors

In this section, I define a function that trains a multi-layer perceptron by randomly initializing the network’s weights and biases, then applying backpropagation with momentum. At each epoch, I propagate inputs forward to compute the output and error, then propagate gradients backward to update the weights and biases.

def backprop_random(a, b, hidden_layers=[5], epochs=1000, eta=0.01, momentum=0.9):
    """
    Trains a multi-layer perceptron using random weight initialization and backpropagation with momentum.

    Parameters:
        a (np.ndarray): Input data of shape (input_neurons, number of samples).
        b (np.ndarray): Target output data of shape (output_neurons, number of samples).
        hidden_layers (list): List specifying the number of neurons in each hidden layer.
        epochs (int): Number of training iterations.
        eta (float): Learning rate.
        momentum (float): Momentum factor for weight updates.

    Returns:
        error_list (list): List containing the mean squared error for each epoch.
        weights (list): List of weight matrices for each layer.
        biases (list): List of bias vectors for each layer.
    """
    input_neurons = np.shape(a)[0]
    output_neurons = np.shape(b)[0]

    # Initialize weights and biases
    layer_sizes = [input_neurons] + hidden_layers + [output_neurons]
    weights = [np.random.uniform(low=-1, high=1, size=(layer_sizes[i+1], layer_sizes[i]))
               for i in range(len(layer_sizes)-1)]
    biases = [np.random.uniform(low=-1, high=1, size=(size, 1)) for size in layer_sizes[1:]]

    # Initialize momentum deltas
    v_weights = [np.zeros(w.shape) for w in weights]
    v_biases = [np.zeros(b.shape) for b in biases]

    error_list = []

    for i in range(epochs):
        # Forward propagation
        net_inputs = [np.dot(weights[0], a) + biases[0]]
        outputs = [sigmoid(net_inputs[0])]
        for j in range(1, len(weights)):
            net_inputs.append(np.dot(weights[j], outputs[j-1]) + biases[j])
            outputs.append(sigmoid(net_inputs[j]))

        # Compute output error
        error = outputs[-1] - b
        error_list.append(np.mean(error**2))

        # Backpropagation
        delta = [error * d_sigmoid(net_inputs[-1])]
        for j in range(len(weights) - 2, -1, -1):
            delta.insert(0, np.dot(weights[j + 1].T, delta[0]) * d_sigmoid(net_inputs[j]))

        # Compute gradients for weights and biases
        w_delta = [np.dot(delta[0], a.T)]
        b_delta = [np.sum(delta[0], axis=1, keepdims=True)]
        for j in range(1, len(weights)):
            w_delta.append(np.dot(delta[j], outputs[j - 1].T))
            b_delta.append(np.sum(delta[j], axis=1, keepdims=True))

        # Update weights and biases with momentum
        for j in range(len(weights)):
            v_weights[j] = momentum * v_weights[j] + eta * w_delta[j]
            weights[j] -= v_weights[j]
            v_biases[j] = momentum * v_biases[j] + eta * b_delta[j]
            biases[j] -= v_biases[j]

    return error_list, weights, biases
